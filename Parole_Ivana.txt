using System;
using System.Security.Cryptography; // MD5
using System.Text;
using System.Threading.Tasks;       // task-uri si Parallel.ForEach
using System.Diagnostics;           // stopwatch
using System.Threading;             // Thread ID

namespace GeneratorParoleMD5
{
    class Program
    {
        private static readonly string Alfabet = "abcdefghijklmnopqrstuvwxyz";
        private static readonly int LungimeParola = 4;

        static void Main(string[] args)
        {
            string[] hashUri = {
                "56aed7e7485ff03d5605b885b86e947e",
                "e26026b73cdc3b59012c318ba26b5518",
                "9de37a0627c25684fdd519ca84073e34"
            };

            // --- ETAPA 1: Executie pe un singur fir (Single-thread) ---
            RulareEtapaSecventiala(hashUri);

            Console.WriteLine("\n\nAPASATI O TASTA PENTRU A TRECE LA ETAPA 2 (FIRE DE EXECUTIE)...\n");
            Console.ReadKey();

            // --- ETAPA 2: Executie paralela pe mai multe fire (Cerinta 2) ---
            RulareEtapaParalela(hashUri);

            Console.WriteLine("\n==================================================");
            Console.WriteLine("   PROIECT FINALIZAT");
            Console.WriteLine("==================================================");
            Console.ReadKey();
        }

        // met care proceseaza hash-urile unul cate unul, in mod secvential
        private static void RulareEtapaSecventiala(string[] hashUri)
        {
            Console.WriteLine("==================================================");
            Console.WriteLine("   ETAPA 1: DECRIPTARE CLASICA (SINGLE-THREAD)   ");
            Console.WriteLine("==================================================");

            foreach (var hash in hashUri)
            {
                Console.WriteLine("[*] Analiza Hash: " + hash);
                Stopwatch sw = Stopwatch.StartNew();
                
                // init un vector de caractere gol si pornim backtracking-ul de la indexul 0
                string rezultat = BacktrackingClasic(new char[LungimeParola], 0, hash);
                
                sw.Stop();
                Console.WriteLine("[OK] Gasit: " + rezultat + " | Timp: " + sw.ElapsedMilliseconds + " ms\n");
            }
        }

        // met care proceseaza hash-urile utilizand paralelizarea
        private static void RulareEtapaParalela(string[] hashUri)
        {
            Console.WriteLine("==================================================");
            Console.WriteLine("   ETAPA 2: DECRIPTARE OPTIMIZATA (MULTITHREADING)");
            Console.WriteLine("==================================================");

            foreach (var hash in hashUri)
            {
                Console.WriteLine("[*] Analiza Hash: " + hash);
                Stopwatch sw = Stopwatch.StartNew();

                string rezultat = DecripteazaCuMaiMulteFire(hash);

                sw.Stop();
                Console.WriteLine("[OK] Gasit: " + rezultat + " | Timp: " + sw.ElapsedMilliseconds + " ms\n");
            }
        }

        // REZOLVARE CERINTA 2: Administrarea firelor de executie
        private static string DecripteazaCuMaiMulteFire(string hashTinta)
        {
            string rezultatFinal = null;

            // Parallel.ForEach imparte alfabetul pe firele de executie disponibile
            // fiecare litera din alfabet va fi un punct de pornire pentru un fir separat
            Parallel.ForEach(Alfabet, (litera, stare) =>
            {
                // vectorul solutie pentru firul curent
                char[] vector = new char[LungimeParola];
                vector[0] = litera; 
                
                // obtin ID-ul unic al firului de executie curent pentru monitorizare
                int idFir = Thread.CurrentThread.ManagedThreadId;

                // backtracking-ul de la indexul 1 
                string gasit = BacktrackingClasic(vector, 1, hashTinta);

                if (gasit != null)
                {
                    rezultatFinal = gasit;
                    Console.WriteLine("    [INFO] Firul de executie ID " + idFir + " a terminat sarcina.");

                    stare.Stop();
                }
            });

            return rezultatFinal;
        }

        // Backtracking pentru explorarea spatiului solutiilor
        private static string BacktrackingClasic(char[] parola, int k, string hashTinta)
        {
            // cond de oprire: am completat toate cele 4 pozitii ale parolei
            if (k == LungimeParola)
            {
                string textClar = new string(parola);
                // CERINTA 1: Verificam daca hash-ul textului generat corespunde cu hash-ul tinta
                if (CalculMD5(textClar) == hashTinta) return textClar;
                return null;
            }

            // iterez prin toate caracterele posibile din alfabet pentru pozitia curenta 'k'
            for (int i = 0; i < Alfabet.Length; i++)
            {
                parola[k] = Alfabet[i]; // Alegem o litera
                
                // apelez recursiv functia pentru urmatoarea pozitie (k + 1)
                string gasit = BacktrackingClasic(parola, k + 1, hashTinta);
                
                // daca ramura de recursivitate a gasit parola, o returnam imediat in sus
                if (gasit != null) return gasit;
            }
            
            return null; 
        }

        // REZOLVARE CERINTA 1: Metoda de hashing MD5
        private static string CalculMD5(string input)
        {
            // istantiez algoritmul MD5 furnizat de .NET
            using (MD5 md5 = MD5.Create())
            {
                // conv textul clar in array de bytes (ASCII)
                byte[] bytes = md5.ComputeHash(Encoding.ASCII.GetBytes(input));
                
                // conv array-ul de bytes inapoi intr-un string Hexadecimal
                StringBuilder sb = new StringBuilder();
                foreach (byte b in bytes)
                {
                    // "x2" formateaza byte-ul ca string hexa de 2 caractere (ex: 0F)
                    sb.Append(b.ToString("x2"));
                }
                return sb.ToString();
            }
        }
    }
}
